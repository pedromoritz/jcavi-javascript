Date():

.getYear(): (anos desde 1900)

.getFullYear(): (ano completo)

.getDate(): (dia)

.getMonth(): (mês, onde 0 é Janeiro e 11 é Dezembro)

.getDay(): (dia da semana, onde 0 é Domingo e 6 é Sábado)

.getHours(): (horas)

.getMinutes(): (minutos)

.getSeconds(): (segundos)
